### エンディング到達時に記録する

各エンディングの最後でこちらの1行を足す。
```
    # ...エンディング...
    $ mark_ending("A")
```

1) クリア記録（persistent）を用意（安全な初期化込み）

新規に system_flags.rpy ファイルを作る。
# ------------------------------------------------------------
# system_flags.rpy
# クリア状態を「周回をまたいで」保存する（persistent）
# ------------------------------------------------------------

init python:
    # 初回起動時は persistent に項目が無いので作る
    if not hasattr(persistent, "endings_cleared"):
        persistent.endings_cleared = []  # 例: ["A", "C"]

    def mark_ending(ending_id: str) -> None:
        """
        クリア済みエンディングを persistent に記録する。

        ending_id:
            "A", "B", "C" のような短いIDを想定。
        """
        # 念のため入れ物チェック（事故防止）
        if not hasattr(persistent, "endings_cleared"):
            persistent.endings_cleared = []

        # 二重登録防止
        if ending_id not in persistent.endings_cleared:
            persistent.endings_cleared.append(ending_id)

        # すぐ書き込み（テスト中も安心）
        renpy.save_persistent()

# ------------------------------------------------------------
# persistent の安全な初期化
# ------------------------------------------------------------
init python:
    # 属性が無い or None ならリストで作る
    if (not hasattr(persistent, "endings_cleared")) or (persistent.endings_cleared is None):
        persistent.endings_cleared = []
        renpy.save_persistent()

init python:
    def reset_clears():
        """
        クリア済みエンディング記録を全消去する（テスト用）
        """        
        persistent.endings_cleared = []
        renpy.save_persistent()
        renpy.notify("クリア記録をリセットしました。")

## Navigation（ナビゲーション）スクリーン ####################################################
##
## このスクリーンはメインメニューとゲームメニューに表示され、各メニュー間を移
## 動したり、ゲームをスタートしたりする機能を提供しています。

screen navigation():

    vbox:
        style_prefix "navigation"

        xpos gui.navigation_xpos
        yalign 0.5

        spacing gui.navigation_spacing

        if main_menu:

            textbutton _("スタート") action Start()
            # その場で関数を呼ぶ（本編に入らない）
            # ※ 本番前にこのボタンは消してね 👍
            textbutton _("【デバッグ】クリア初期化") action Function(reset_clears)
            textbutton _("エンディング一覧") action ShowMenu("ending_list_screen")

            # ------------------------------------------------------------
            # ★解放メニュー（タイトル画面でのみ表示）
            # persistent.endings_cleared が None の場合でも落ちないように防御
            # ------------------------------------------------------------

            # persistent.endings_cleared を安全に取り出す
            $ _clears = getattr(persistent, "endings_cleared", [])

            # もし None や変な型が入っていたら、空リスト扱いにする
            if _clears is None:
                $ _clears = []

            # ヒント：1つでもクリア済みなら解放
            if len(_clears) >= 1:
                textbutton _("ヒント") action ShowMenu("hint_screen")
            else:
                textbutton _("？？？") action NullAction()

            # あとがき：3エンディング全部クリアで解放
            if len(_clears) >= 3:
                textbutton _("あとがき") action ShowMenu("afterword_screen")
            else:
                textbutton _("？？？") action NullAction()
        else:

            textbutton _("ヒストリー") action ShowMenu("history")

            textbutton _("セーブ") action ShowMenu("save")

        textbutton _("ロード") action ShowMenu("load")

        textbutton _("環境設定") action ShowMenu("preferences")

        if _in_replay:

            textbutton _("リプレイ終了") action EndReplay(confirm=True)

        elif not main_menu:

            textbutton _("メインメニュー") action MainMenu()

        textbutton _("バージョン情報") action ShowMenu("about")

        if renpy.variant("pc") or (renpy.variant("web") and not renpy.variant("mobile")):

            ## モバイルデバイスにはヘルプは不要であるか不適切です。
            textbutton _("ヘルプ") action ShowMenu("help")

        if renpy.variant("pc"):

            ## 終了ボタンはiOSでは使用できません。また、AndroidやWebでは必要あ
            ## りません。
            textbutton _("終了") action Quit(confirm=not main_menu)



## Main Menu（メインメニュー）スクリーン #####################################################
##
## Ren'Py が起動した時に表示されるメインメニューを表示するスクリーンです。
##
## https://ja.renpy.org/doc/html/screen_special.html#main-menu

# --------------------------------------
# Transform を使って「フェードイン」
# --------------------------------------
# 黒幕をフェードアウトして、下の背景を見せる
transform fade_out_black:
    alpha 1.0
    linear 0.35 alpha 0.0

# --------------------------------------


screen main_menu():

    ## 次のコードは、同じタグを持つ他のメニュースクリーンが表示された時にスク
    ## リーンを置換します。
    tag menu

    # add gui.main_menu_background # ← 二重になるので基本は消すらしい/コメントアウト

    ## 次の空のフレームは gui/overlay/main_menu.png を表示してメインメニューを
    ## 暗くしています。
    frame:
        style "main_menu_frame"

    ## use ステートメントは、他のスクリーンを現在のスクリーンの内に表示するの
    ## に使います。メインメニューの実際のコンテンツは navigation（ナビゲーショ
    ## ン）スクリーンです。

    # ------------------------------------------------------------
    # ★ タイトル背景（通常 / 全クリア切替）
    # ------------------------------------------------------------


    $ clears = getattr(persistent, "endings_cleared", []) or []

    # 1) 背景画像は最初から「不透明で」出しておく
    if all(e in clears for e in ["A", "B", "C"]):
        add "title_bg_clear.png" xysize (config.screen_width, config.screen_height)
    else:
        add "title_bg_normal.png" xysize (config.screen_width, config.screen_height)

    # 2) その上に黒幕を置いて、黒幕だけフェードアウト
    add Solid("#000") at fade_out_black xysize (config.screen_width, config.screen_height)

    use navigation # この use navigation の少し上が、背景を置く最高の場所なんだそうな。

    if gui.show_name:

        vbox:
            style "main_menu_vbox"

            text "[config.name!t]":
                style "main_menu_title"

            text "[config.version]":
                style "main_menu_version"

# ------------------------------------------------------------
# ヒント画面（メインメニュー用）
# ・tag menu を付けると「メニューとして」扱われます（重要）
# ・Return() で元のメインメニューに戻れます
# ------------------------------------------------------------
screen hint_screen():
    tag menu

    add Solid("#000")  # 仮の背景。お好みで消してOK

    frame:
        xalign 0.5
        yalign 0.5
        xsize 0.8
        ysize 0.7

        vbox:
            spacing 20

            text "ヒント" size 48
            text "（仮）ヒントです。" size 28

            textbutton "戻る" action Return()

# ------------------------------------------------------------
# エンディング一覧（？？？表示つき）
# ・メインメニューから ShowMenu で開く想定
# ・persistent.endings_cleared に "A","B","C" が入る想定
# ------------------------------------------------------------
screen ending_list_screen():
    tag menu

    # クリア情報を安全に取得（None対策込み）
    $ clears = getattr(persistent, "endings_cleared", [])
    if clears is None:
        $ clears = []

    add Solid("#000")

    frame:
        xalign 0.5
        yalign 0.5
        xsize 0.85
        ysize 0.80

        vbox:
            spacing 18

            text "エンディング一覧" size 48

            # 表示用（A/B/C の想定）
            $ ending_defs = [
                ("A", "エンディングA：ノーマル"),
                ("B", "エンディングB：グッド"),
                ("C", "エンディングC：ベスト"),
            ]

            # -----------------------------
            # エンディング一覧
            # -----------------------------
            for eid, title in ending_defs:

                if eid in clears:
                    hbox:
                        spacing 16
                        text "CLEAR" size 26
                        text title size 30
                else:
                    hbox:
                        spacing 16
                        text "----" size 26
                        text "？？？" size 30

            null height 20

            # -----------------------------
            # ★ 全エンディング達成メッセージ
            # -----------------------------
            if all(e in clears for e in ["A", "B", "C"]):
                text "全エンディング達成おめでとう！" size 26

            null height 10

            # デバッグ用
            text "クリア状況: [', '.join(clears) if clears else '-']" size 20

            null height 10

            textbutton "戻る" action Return()
